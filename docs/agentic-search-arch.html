async function agentLoop(userMessage) {
  const messages = [
    { role: 'system', content: SYSTEM_PROMPT },
    { role: 'user',   content: userMessage }
  ];
  let round = 0;
  while (round < MAX_ROUNDS) {
    round++;
    // 1. 
    const reply = await callLLM(messages);
    // 2. 
    const toolCall = parseToolCall(reply);
    if (!toolCall) {
      // 
      return reply;
    }
    // 3. 
    messages.push({ role: 'assistant', content: reply });
    // 4. 
    const result = await executeTool(toolCall);
    // 5. 
    messages.push({ role: 'user', content: `<tool_result tool="${toolCall.tool}">\n${result}\n</tool_result>` });
    // 6. 
    if (ctxSize(messages) > CTX_LIMIT)
      messages = await summarizeContext(messages, userMessage);
  }
  // 
  messages.push({ role: 'user', content:  });
  return callLLM(messages);
}

const TOOL_REGEX = /<tool_call>\s*([\s\S]*?)\s*<\/tool_call>/i;

function parseToolCall(text) {
  const match = text.match(TOOL_REGEX);
  if (!match) return null;
  try { return JSON.parse(match[1]); }
  catch {
    // 
    return parseToolCallFallback(match[1]);
  }
}

// 
function parseToolCallFallback(raw) {
  const tm = raw.match(/"tool"\s*:\s*"([^"]+)"/);
  const qm = raw.match(/"query"\s*:\s*"([^"]+)"/);
  const um = raw.match(/"url"\s*:\s*"([^"]+)"/);
  if (!tm) return null;
  return { tool: tm[1], args: { ...(qm && { query: qm[1] }), ...(um && { url: um[1] }) } };
}

// 
function parseAllToolCalls(text) {
  const re = /<tool_call>\s*([\s\S]*?)\s*<\/tool_call>/gi;
  const calls = []; let m;
  while ((m = re.exec(text)) !== null)
    try { calls.push(JSON.parse(m[1])); } catch {}
  return calls; // 
}

const TOOLS = {
  async web_search({ query }) {
    const res = await searchEngine.query(query);
    return res.slice(0,5).map(r=>`[${r.title}](${r.url})\n${r.snippet}`).join('\n\n');
  },
  async read_page({ url }) {
    const html = await fetch(url).then(r=>r.text());
    return extractText(html).slice(0,8000); // 
  },
  async calc({ expression }) {
    try { return String(eval(expression)); } // 
    catch(e) { return `error: ${e.message}`; }
  }
};
async function executeTool({ tool, args }) {
  const fn = TOOLS[tool];
  if (!fn) return `ERROR: '${tool}' : ${Object.keys(TOOLS).join(', ')}`;
  try {
    return `##  ${tool}\n\n${await fn(args)}`;
  } catch(e) {
    // 
    return `ERROR  ${tool}: ${e.message}`;
  }
}

  //

    .map(m=>`[${m.role}]: ${m.content}`).join('\n\n');

    { role:'user',   content: prompt }

    { role:'assistant', content: `[]\n${summary}` },

function extractEntities(text) {
  return {
    tickers: [...text.matchAll(/\b[A-Z]{2,5}\b/g)].map(m=>m[0]),
    dates:   [...text.matchAll(/\d{4}-\d{2}-\d{2}/g)].map(m=>m[0]),
    urls:    [...text.matchAll(/(https?:\/\/[^\s]+)/g)].map(m=>m[1]),
    amounts: [...text.matchAll(/\$[\d,]+(\.\d{2})?/g)].map(m=>m[0])
  };
}
